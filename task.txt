Copy to clipboard

/// POST /api/manual-trade — Submit a manual trade from the dashboard.
///
/// The trade is validated and then injected into the execution pipeline
/// exactly like an automated trade: pre-trade risk, position slots,
/// SL/TP conditional orders, exit evaluator tracking, etc.
async fn api_manual_trade(
    State(state): State<Arc<AppState>>,
    Json(req): Json<ManualTradeRequest>,
) -> Json<ManualTradeResponse> {
    // Basic validation
    if req.symbol.is_empty() {
        return Json(ManualTradeResponse {
            success: false,
            message: "Symbol is required".to_string(),
            order_id: None,
        });
    }
    
    let side_lower = req.side.to_lowercase();
    if side_lower != "buy" && side_lower != "sell" {
        return Json(ManualTradeResponse {
            success: false,
            message: "Side must be 'buy' or 'sell'".to_string(),
            order_id: None,
        });
    }
    
    if req.size <= 0 {
        return Json(ManualTradeResponse {
            success: false,
            message: "Size must be positive".to_string(),
            order_id: None,
        });
    }
    
    if req.stop_loss <= 0.0 || req.take_profit <= 0.0 {
        return Json(ManualTradeResponse {
            success: false,
            message: "Stop loss and take profit are required".to_string(),
            order_id: None,
        });
    }
    
    // Validate SL/TP direction
    if side_lower == "buy" && req.stop_loss >= req.take_profit {
        return Json(ManualTradeResponse {
            success: false,
            message: "For BUY: stop_loss must be below take_profit".to_string(),
            order_id: None,
        });
    }
    if side_lower == "sell" && req.stop_loss <= req.take_profit {
        return Json(ManualTradeResponse {
            success: false,
            message: "For SELL: stop_loss must be above take_profit".to_string(),
            order_id: None,
        });
    }
    
    // Send to execution pipeline
    match state.manual_trade_tx.send(req.clone()).await {
        Ok(()) => {
            info!("[dashboard] Manual trade submitted: {} {} {} contracts SL={} TP={}",
                req.side, req.symbol, req.size, req.stop_loss, req.take_profit);
            Json(ManualTradeResponse {
                success: true,
                message: format!("Manual trade submitted: {} {} {} contracts", req.side, req.symbol, req.size),
                order_id: None, // Will be filled when execution confirms
            })
        }
        Err(e) => {
            Json(ManualTradeResponse {
                success: false,
                message: format!("Failed to submit trade: {}", e),
                order_id: None,
            })
        }
    }
}

function previewManualTrade() {
    const symbol = document.getElementById('mt-symbol').value;
    const side = document.getElementById('mt-side').value;
    const size = document.getElementById('mt-size').value;
    const leverage = document.getElementById('mt-leverage').value;
    const price = document.getElementById('mt-price').value;
    const sl = document.getElementById('mt-sl').value;
    const tp = document.getElementById('mt-tp').value;
    
    if (!sl || !tp) {
        alert('Stop Loss and Take Profit are required!');
        return;
    }
    
    const priceStr = price ? `@ ${price}` : '@ MARKET';
    const rr = sl && tp && price ? 
        Math.abs((parseFloat(tp) - parseFloat(price)) / (parseFloat(price) - parseFloat(sl))).toFixed(2) : 
        'N/A';
    
    const preview = document.getElementById('mt-preview');
    preview.className = `alert ${side === 'buy' ? 'alert-success' : 'alert-danger'}`;
    preview.innerHTML = `
        <strong>${side.toUpperCase()} ${symbol}</strong> — ${size} contracts ${priceStr}<br>
        Leverage: ${leverage}x | SL: ${sl} | TP: ${tp} | R:R = 1:${rr}<br>
        <small class="text-muted">This trade will be managed by the engine (auto SL/TP, trailing stop, risk management)</small>
    `;
    preview.classList.remove('d-none');
}

document.getElementById('manualTradeFormEl').addEventListener('submit', async function(e) {
    e.preventDefault();
    
    const submitBtn = document.getElementById('mt-submit-btn');
    const statusEl = document.getElementById('mt-status');
    
    submitBtn.disabled = true;
    statusEl.textContent = 'Submitting...';
    
    const priceVal = document.getElementById('mt-price').value;
    const payload = {
        symbol: document.getElementById('mt-symbol').value,
        side: document.getElementById('mt-side').value,
        size: parseInt(document.getElementById('mt-size').value),
        leverage: parseInt(document.getElementById('mt-leverage').value),
        price: priceVal ? parseFloat(priceVal) : null,
        stop_loss: parseFloat(document.getElementById('mt-sl').value),
        take_profit: parseFloat(document.getElementById('mt-tp').value),
        exchange: document.getElementById('mt-exchange').value,
        note: document.getElementById('mt-note').value || null,
    };
    
    try {
        const resp = await fetch('/api/manual-trade', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(payload),
        });
        const data = await resp.json();
        
        if (data.success) {
            statusEl.innerHTML = `<span class="text-success">✅ ${data.message}</span>`;
            // Reset form
            document.getElementById('mt-price').value = '';
            document.getElementById('mt-sl').value = '';
            document.getElementById('mt-tp').value = '';
            document.getElementById('mt-note').value = '';
            document.getElementById('mt-preview').classList.add('d-none');
        } else {
            statusEl.innerHTML = `<span class="text-danger">❌ ${data.message}</span>`;
        }
    } catch (err) {
        statusEl.innerHTML = `<span class="text-danger">❌ Network error: ${err.message}</span>`;
    } finally {
        submitBtn.disabled = false;
    }
});

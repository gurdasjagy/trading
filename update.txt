```rust
//! Hardware timestamp support for ultra-low latency tick timing.
//!
//! Uses SO_TIMESTAMPING on Linux to get NIC-level timestamps instead of
//! kernel timestamps, reducing timestamp jitter from ~50μs to ~100ns.

use std::io;
use std::os::unix::io::RawFd;

/// Enable hardware timestamps on a socket.
/// Call this after socket creation but before binding.
#[cfg(target_os = "linux")]
pub fn enable_hw_timestamp(fd: RawFd) -> io::Result<()> {
    use libc::{
        setsockopt, SOL_SOCKET, SO_TIMESTAMPING,
        SOF_TIMESTAMPING_RX_HARDWARE, SOF_TIMESTAMPING_TX_HARDWARE,
        SOF_TIMESTAMPING_RAW_HARDWARE,
    };
    
    let flags: u32 = SOF_TIMESTAMPING_RX_HARDWARE 
        | SOF_TIMESTAMPING_TX_HARDWARE 
        | SOF_TIMESTAMPING_RAW_HARDWARE;
    
    let ret = unsafe {
        setsockopt(
            fd,
            SOL_SOCKET,
            SO_TIMESTAMPING,
            &flags as *const u32 as *const libc::c_void,
            std::mem::size_of::<u32>() as libc::socklen_t,
        )
    };
    
    if ret < 0 {
        Err(io::Error::last_os_error())
    } else {
        Ok(())
    }
}

#[cfg(not(target_os = "linux"))]
pub fn enable_hw_timestamp(_fd: RawFd) -> io::Result<()> {
    Ok(()) // No-op on non-Linux
}
```

```rust
//! High-performance tick database using memory-mapped append-only logs.
//!
//! Format: Each day creates a new file: /data/ticks/YYYY-MM-DD/{symbol}.tick
//! Each tick is 48 bytes: timestamp(8) + price(8) + qty(8) + side(1) + pad(23)
//!
//! For replay: mmap the file read-only and iterate at desired speed.

use std::fs::{File, OpenOptions};
use std::io::{self, Write};
use std::path::PathBuf;
use memmap2::{MmapMut, MmapOptions};

const TICK_SIZE: usize = 48;
const PREALLOC_TICKS: usize = 10_000_000; // ~480MB per file

#[repr(C, packed)]
#[derive(Clone, Copy, Debug)]
pub struct StoredTick {
    pub timestamp_ns: u64,
    pub price_fp: i64,
    pub qty_fp: i64,
    pub side: u8,      // 0=bid, 1=ask
    pub flags: u8,     // 1=trade, 2=book_update, 4=snapshot_marker
    pub _pad: [u8; 22],
}

pub struct TickWriter {
    mmap: MmapMut,
    offset: usize,
    path: PathBuf,
}

impl TickWriter {
    pub fn new(symbol: &str, date: &str) -> io::Result<Self> {
        let dir = PathBuf::from(format!("/data/ticks/{}", date));
        std::fs::create_dir_all(&dir)?;
        
        let path = dir.join(format!("{}.tick", symbol));
        let file = OpenOptions::new()
            .read(true)
            .write(true)
            .create(true)
            .open(&path)?;
        
        file.set_len((TICK_SIZE * PREALLOC_TICKS) as u64)?;
        
        let mmap = unsafe { MmapMut::map_mut(&file)? };
        
        Ok(Self { mmap, offset: 0, path })
    }
    
    #[inline]
    pub fn write(&mut self, tick: &StoredTick) -> io::Result<()> {
        if self.offset + TICK_SIZE > self.mmap.len() {
            return Err(io::Error::new(io::ErrorKind::Other, "Tick file full"));
        }
        
        let bytes = unsafe {
            std::slice::from_raw_parts(
                tick as *const StoredTick as *const u8,
                TICK_SIZE
            )
        };
        
        self.mmap[self.offset..self.offset + TICK_SIZE].copy_from_slice(bytes);
        self.offset += TICK_SIZE;
        Ok(())
    }
    
    pub fn sync(&self) -> io::Result<()> {
        self.mmap.flush()
    }
}

pub struct TickReader {
    data: memmap2::Mmap,
    offset: usize,
    len: usize,
}

impl TickReader {
    pub fn open(symbol: &str, date: &str) -> io::Result<Self> {
        let path = PathBuf::from(format!("/data/ticks/{}/{}.tick", date, symbol));
        let file = File::open(&path)?;
        let data = unsafe { MmapOptions::new().map(&file)? };
        let len = data.len() / TICK_SIZE;
        Ok(Self { data, offset: 0, len })
    }
    
    pub fn next(&mut self) -> Option<StoredTick> {
        if self.offset >= self.len {
            return None;
        }
        
        let byte_offset = self.offset * TICK_SIZE;
        let tick: StoredTick = unsafe {
            std::ptr::read(self.data[byte_offset..].as_ptr() as *const StoredTick)
        };
        self.offset += 1;
        Some(tick)
    }
    
    pub fn seek(&mut self, tick_index: usize) {
        self.offset = tick_index.min(self.len);
    }
    
    pub fn len(&self) -> usize {
        self.len
    }
}
```

```rust
//! Real-Time Value at Risk Engine using Historical Simulation.
//!
//! Maintains a rolling window of returns and computes VaR at 95%/99% confidence.
//! Updates on every position change.

use std::collections::VecDeque;

const RETURN_WINDOW: usize = 1000; // ~16 hours of 1-minute returns
const TRADING_DAYS: f64 = 252.0;

#[derive(Debug, Clone)]
pub struct VaRMetrics {
    pub var_95: f64,      // 95% VaR in USDT
    pub var_99: f64,      // 99% VaR in USDT
    pub cvar_95: f64,     // Conditional VaR (Expected Shortfall) at 95%
    pub cvar_99: f64,     // Conditional VaR at 99%
    pub marginal_var: f64, // Marginal VaR for next position
}

pub struct VaREngine {
    /// Rolling window of portfolio returns (as percentage)
    returns: VecDeque<f64>,
    /// Current portfolio value in USDT
    portfolio_value: f64,
    /// Cached sorted returns for percentile calculation
    sorted_returns: Vec<f64>,
    /// Flag indicating sorted_returns needs update
    dirty: bool,
    /// Per-symbol volatility estimates
    symbol_volatilities: [f64; 16], // Up to 16 symbols
}

impl VaREngine {
    pub fn new() -> Self {
        Self {
            returns: VecDeque::with_capacity(RETURN_WINDOW),
            portfolio_value: 0.0,
            sorted_returns: Vec::with_capacity(RETURN_WINDOW),
            dirty: true,
            symbol_volatilities: [0.02; 16], // Default 2% daily vol
        }
    }
    
    /// Record a new return observation.
    /// `return_pct` is the portfolio return for the period (e.g., 0.001 = 0.1%)
    pub fn record_return(&mut self, return_pct: f64) {
        if self.returns.len() >= RETURN_WINDOW {
            self.returns.pop_front();
        }
        self.returns.push_back(return_pct);
        self.dirty = true;
    }
    
    /// Update portfolio value for VaR calculation
    pub fn set_portfolio_value(&mut self, value: f64) {
        self.portfolio_value = value;
    }
    
    /// Update volatility estimate for a symbol
    pub fn update_symbol_volatility(&mut self, symbol_id: u16, daily_vol: f64) {
        if (symbol_id as usize) < self.symbol_volatilities.len() {
            self.symbol_volatilities[symbol_id as usize] = daily_vol;
        }
    }
    
    /// Compute VaR metrics using historical simulation
    pub fn compute(&mut self) -> VaRMetrics {
        if self.returns.len() < 30 {
            // Insufficient data - return conservative estimate
            return VaRMetrics {
                var_95: self.portfolio_value * 0.05,
                var_99: self.portfolio_value * 0.10,
                cvar_95: self.portfolio_value * 0.07,
                cvar_99: self.portfolio_value * 0.12,
                marginal_var: self.portfolio_value * 0.02,
            };
        }
        
        if self.dirty {
            self.sorted_returns = self.returns.iter().cloned().collect();
            self.sorted_returns.sort_by(|a, b| a.partial_cmp(b).unwrap());
            self.dirty = false;
        }
        
        let n = self.sorted_returns.len();
        
        // VaR at 95% (5th percentile of losses)
        let idx_95 = (n as f64 * 0.05).floor() as usize;
        let var_95_pct = -self.sorted_returns[idx_95].min(0.0);
        
        // VaR at 99% (1st percentile of losses)
        let idx_99 = (n as f64 * 0.01).floor() as usize;
        let var_99_pct = -self.sorted_returns[idx_99].min(0.0);
        
        // CVaR: Average of returns below VaR threshold
        let cvar_95_pct: f64 = self.sorted_returns[..=idx_95]
            .iter()
            .map(|r| -r.min(&0.0))
            .sum::<f64>() / (idx_95 + 1) as f64;
        
        let cvar_99_pct: f64 = self.sorted_returns[..=idx_99]
            .iter()
            .map(|r| -r.min(&0.0))
            .sum::<f64>() / (idx_99 + 1) as f64;
        
        // Scale from 1-period to 1-day (sqrt(N) rule)
        // Assuming 1-minute returns, scale to daily
        let scale_factor = (TRADING_DAYS / RETURN_WINDOW as f64).sqrt();
        
        VaRMetrics {
            var_95: self.portfolio_value * var_95_pct * scale_factor,
            var_99: self.portfolio_value * var_99_pct * scale_factor,
            cvar_95: self.portfolio_value * cvar_95_pct * scale_factor,
            cvar_99: self.portfolio_value * cvar_99_pct * scale_factor,
            marginal_var: self.portfolio_value * var_95_pct * 0.1, // Simplified
        }
    }
    
    /// Check if a proposed position would breach VaR limits
    pub fn check_position(&mut self, additional_exposure: f64, limit_pct: f64) -> bool {
        let metrics = self.compute();
        let proposed_var = metrics.var_99 * (1.0 + additional_exposure / self.portfolio_value);
        proposed_var <= self.portfolio_value * limit_pct
    }
}
```

```rust
//! Options Greeks calculator with Black-Scholes model.
//!
//! Computes delta, gamma, theta, vega for options positions.
//! Used for gamma exposure and hedging calculations.

use std::f64::consts::{E, PI};

/// Standard normal CDF approximation (Abramowitz and Stegun)
fn norm_cdf(x: f64) -> f64 {
    let a1 = 0.254829592;
    let a2 = -0.284496736;
    let a3 = 1.421413741;
    let a4 = -1.453152027;
    let a5 = 1.061405429;
    let p = 0.3275911;
    
    let sign = if x < 0.0 { -1.0 } else { 1.0 };
    let x = x.abs();
    let t = 1.0 / (1.0 + p * x);
    let y = 1.0 - (((((a5 * t + a4) * t) + a3) * t + a2) * t + a1) * t * E.powf(-x * x / 2.0);
    
    0.5 * (1.0 + sign * y)
}

/// Standard normal PDF
fn norm_pdf(x: f64) -> f64 {
    E.powf(-x * x / 2.0) / (2.0 * PI).sqrt()
}

/// Options Greeks result
#[derive(Debug, Clone, Copy)]
pub struct Greeks {
    pub delta: f64,
    pub gamma: f64,
    pub theta: f64,
    pub vega: f64,
    pub rho: f64,
}

/// Calculate d1 parameter for Black-Scholes
fn d1(s: f64, k: f64, t: f64, r: f64, sigma: f64) -> f64 {
    ((s / k).ln() + (r + sigma * sigma / 2.0) * t) / (sigma * t.sqrt())
}

/// Calculate d2 parameter for Black-Scholes
fn d2(d1: f64, sigma: f64, t: f64) -> f64 {
    d1 - sigma * t.sqrt()
}

/// Calculate Greeks for a call option
pub fn call_greeks(spot: f64, strike: f64, time_years: f64, rate: f64, vol: f64) -> Greeks {
    if time_years <= 0.0 || vol <= 0.0 {
        return Greeks { delta: 0.0, gamma: 0.0, theta: 0.0, vega: 0.0, rho: 0.0 };
    }
    
    let d1_val = d1(spot, strike, time_years, rate, vol);
    let d2_val = d2(d1_val, vol, time_years);
    
    let delta = norm_cdf(d1_val);
    let gamma = norm_pdf(d1_val) / (spot * vol * time_years.sqrt());
    let theta = -(spot * norm_pdf(d1_val) * vol) / (2.0 * time_years.sqrt())
        - rate * strike * E.powf(-rate * time_years) * norm_cdf(d2_val);
    let vega = spot * time_years.sqrt() * norm_pdf(d1_val) / 100.0; // Per 1% vol change
    let rho = strike * time_years * E.powf(-rate * time_years) * norm_cdf(d2_val) / 100.0;
    
    Greeks { delta, gamma, theta, vega, rho }
}

/// Calculate Greeks for a put option
pub fn put_greeks(spot: f64, strike: f64, time_years: f64, rate: f64, vol: f64) -> Greeks {
    if time_years <= 0.0 || vol <= 0.0 {
        return Greeks { delta: 0.0, gamma: 0.0, theta: 0.0, vega: 0.0, rho: 0.0 };
    }
    
    let d1_val = d1(spot, strike, time_years, rate, vol);
    let d2_val = d2(d1_val, vol, time_years);
    
    let delta = norm_cdf(d1_val) - 1.0;
    let gamma = norm_pdf(d1_val) / (spot * vol * time_years.sqrt());
    let theta = -(spot * norm_pdf(d1_val) * vol) / (2.0 * time_years.sqrt())
        + rate * strike * E.powf(-rate * time_years) * norm_cdf(-d2_val);
    let vega = spot * time_years.sqrt() * norm_pdf(d1_val) / 100.0;
    let rho = -strike * time_years * E.powf(-rate * time_years) * norm_cdf(-d2_val) / 100.0;
    
    Greeks { delta, gamma, theta, vega, rho }
}

/// Aggregate gamma exposure across strike prices
pub struct GammaExposure {
    /// Net gamma at each price level (strike -> gamma)
    pub levels: Vec<(f64, f64)>,
    /// Gamma flip level (where net gamma changes sign)
    pub flip_level: Option<f64>,
}

impl GammaExposure {
    pub fn compute(spot: f64, positions: &[(f64, f64, f64, bool)]) -> Self {
        // positions: (strike, quantity, time_years, is_call)
        let mut levels: Vec<(f64, f64)> = Vec::new();
        
        let vol = 0.5; // Simplified: use 50% implied vol
        let rate = 0.05; // 5% risk-free rate
        
        for &(strike, qty, time, is_call) in positions {
            let greeks = if is_call {
                call_greeks(spot, strike, time, rate, vol)
            } else {
                put_greeks(spot, strike, time, rate, vol)
            };
            levels.push((strike, greeks.gamma * qty));
        }
        
        // Sort by strike
        levels.sort_by(|a, b| a.0.partial_cmp(&b.0).unwrap());
        
        // Find gamma flip level
        let mut flip_level = None;
        let mut cumulative = 0.0;
        for (strike, gamma) in &levels {
            let prev_cum = cumulative;
            cumulative += gamma;
            if prev_cum.signum() != cumulative.signum() && cumulative.signum() != 0.0 {
                flip_level = Some(*strike);
                break;
            }
        }
        
        Self { levels, flip_level }
    }
}
```

```rust
//! Multi-Exchange Arbitrage Engine.
//!
//! Monitors price disparities across exchanges and executes simultaneous
//! buy/sell orders when spread exceeds threshold.

use std::collections::HashMap;
use std::sync::atomic::{AtomicU64, Ordering};
use std::sync::Arc;

const ARB_THRESHOLD_BPS: i64 = 10; // 0.1% minimum spread to arb
const MAX_ARB_SIZE_USDT: f64 = 10000.0;

#[derive(Debug, Clone, Copy)]
pub struct ExchangePrice {
    pub exchange_id: u8,
    pub bid: i64,       // FixedPrice format
    pub ask: i64,
    pub bid_size: i64,  // FixedQty format
    pub ask_size: i64,
    pub timestamp_ns: u64,
}

pub struct ArbitrageEngine {
    /// Latest prices per symbol per exchange
    /// Key: (symbol_id, exchange_id) -> ExchangePrice
    prices: HashMap<(u16, u8), ExchangePrice>,
    /// Best bid/ask across all exchanges
    best_bid: HashMap<u16, (i64, u8)>, // (price, exchange_id)
    best_ask: HashMap<u16, (i64, u8)>,
    /// Opportunities detected
    pub opportunities_detected: AtomicU64,
    /// Arbitrage trades executed
    pub trades_executed: AtomicU64,
}

#[derive(Debug, Clone)]
pub struct ArbOpportunity {
    pub symbol_id: u16,
    pub buy_exchange: u8,
    pub sell_exchange: u8,
    pub buy_price: i64,
    pub sell_price: i64,
    pub spread_bps: i64,
    pub max_size: i64,
}

impl ArbitrageEngine {
    pub fn new() -> Self {
        Self {
            prices: HashMap::new(),
            best_bid: HashMap::new(),
            best_ask: HashMap::new(),
            opportunities_detected: AtomicU64::new(0),
            trades_executed: AtomicU64::new(0),
        }
    }
    
    /// Update price from an exchange
    pub fn update_price(&mut self, symbol_id: u16, price: ExchangePrice) {
        self.prices.insert((symbol_id, price.exchange_id), price);
        
        // Update best bid
        let current_best_bid = self.best_bid.get(&symbol_id).map(|&(p, _)| p).unwrap_or(0);
        if price.bid > current_best_bid {
            self.best_bid.insert(symbol_id, (price.bid, price.exchange_id));
        }
        
        // Update best ask
        let current_best_ask = self.best_ask.get(&symbol_id).map(|&(p, _)| p).unwrap_or(i64::MAX);
        if price.ask < current_best_ask {
            self.best_ask.insert(symbol_id, (price.ask, price.exchange_id));
        }
    }
    
    /// Scan for arbitrage opportunities
    pub fn scan(&self, symbol_id: u16) -> Option<ArbOpportunity> {
        let (best_bid, bid_ex) = self.best_bid.get(&symbol_id)?;
        let (best_ask, ask_ex) = self.best_ask.get(&symbol_id)?;
        
        // Can only arb if best bid > best ask (crossed market)
        // Or in practice: best bid on exchange A > best ask on exchange B
        if bid_ex == ask_ex {
            return None; // Same exchange, no arb
        }
        
        if *best_bid <= *best_ask {
            return None; // No arb opportunity
        }
        
        // Calculate spread in basis points
        let mid = (*best_bid + *best_ask) / 2;
        let spread_bps = if mid > 0 {
            (*best_bid - *best_ask) * 10000 / mid
        } else {
            0
        };
        
        if spread_bps < ARB_THRESHOLD_BPS {
            return None;
        }
        
        // Get available sizes
        let bid_price_data = self.prices.get(&(symbol_id, *bid_ex))?;
        let ask_price_data = self.prices.get(&(symbol_id, *ask_ex))?;
```

```rust
/// Intelligent Order Splitting with Participation Rate Adaptation.
///
/// Key improvements over basic TWAP:
/// 1. Adapts to market volume - trades more when volume is high
/// 2. Spread awareness - pauses when spread widens
/// 3. Adverse selection detection - backs off when toxicity rises
/// 4. End-of-period acceleration - completes order within deadline

pub struct AdaptiveTwap {
    /// Total quantity to execute
    pub target_qty: f64,
    /// Executed quantity so far
    pub executed_qty: f64,
    /// Target participation rate (fraction of market volume)
    pub target_participation: f64,
    /// Maximum participation rate
    pub max_participation: f64,
    /// End time (nanoseconds since epoch)
    pub deadline_ns: u64,
    /// Start time
    pub start_ns: u64,
    /// Rolling market volume estimate (per second)
    pub market_volume_per_sec: f64,
    /// Last slice sent
    pub last_slice_ns: u64,
    /// Slices executed
    pub slice_count: u32,
}

impl AdaptiveTwap {
    pub fn new(target_qty: f64, duration_secs: u64, target_participation: f64) -> Self {
        let now = std::time::SystemTime::now()
            .duration_since(std::time::UNIX_EPOCH)
            .unwrap_or_default()
            .as_nanos() as u64;
        
        Self {
            target_qty,
            executed_qty: 0.0,
            target_participation: target_participation.clamp(0.01, 0.25),
            max_participation: 0.30,
            deadline_ns: now + duration_secs * 1_000_000_000,
            start_ns: now,
            market_volume_per_sec: 0.0,
            last_slice_ns: 0,
            slice_count: 0,
        }
    }
    
    /// Update market volume estimate from trade stream
    pub fn update_volume(&mut self, volume_last_second: f64) {
        // Exponential smoothing
        self.market_volume_per_sec = 0.2 * volume_last_second + 0.8 * self.market_volume_per_sec;
    }
    
    /// Calculate next slice size considering market conditions
    pub fn next_slice(&mut self, spread_bps: f64, vpin: f64) -> Option<f64> {
        let now_ns = std::time::SystemTime::now()
            .duration_since(std::time::UNIX_EPOCH)
            .unwrap_or_default()
            .as_nanos() as u64;
        
        // Check if complete
        if self.executed_qty >= self.target_qty {
            return None;
        }
        
        // Check deadline
        if now_ns >= self.deadline_ns {
            // Deadline reached - execute remaining as market
            return Some(self.target_qty - self.executed_qty);
        }
        
        // Spread gating - pause if spread too wide
        if spread_bps > 20.0 {
            return None; // Wait for tighter spread
        }
        
        // VPIN gating - reduce size if flow is toxic
        let vpin_multiplier = if vpin > 0.7 {
            0.25 // Very toxic - trade small
        } else if vpin > 0.5 {
            0.5  // Somewhat toxic
        } else {
            1.0  // Safe
        };
        
        // Time-weighted remaining
        let time_remaining_secs = (self.deadline_ns - now_ns) as f64 / 1e9;
        let qty_remaining = self.target_qty - self.executed_qty;
        let baseline_slice = qty_remaining / (time_remaining_secs / 5.0).max(1.0);
        
        // Volume-adapted slice
        let volume_slice = self.market_volume_per_sec * self.target_participation;
        
        // Take minimum of baseline and volume-adapted, apply VPIN multiplier
        let slice = baseline_slice.min(volume_slice) * vpin_multiplier;
        
        // Ensure minimum slice size
        let min_slice = self.target_qty * 0.01; // 1% minimum
        let final_slice = slice.max(min_slice).min(qty_remaining);
        
        Some(final_slice)
    }
    
    /// Record an execution
    pub fn record_fill(&mut self, qty: f64) {
        self.executed_qty += qty;
        self.slice_count += 1;
        self.last_slice_ns = std::time::SystemTime::now()
            .duration_since(std::time::UNIX_EPOCH)
            .unwrap_or_default()
            .as_nanos() as u64;
    }
    
    /// Get completion percentage
    pub fn progress(&self) -> f64 {
        if self.target_qty > 0.0 {
            (self.executed_qty / self.target_qty * 100.0).min(100.0)
        } else {
            100.0
        }
    }
}
```

```rust
//! Centralized Alert Manager with Rate Limiting and Priority.
//!
//! Aggregates alerts from all components and dispatches via configured channels
//! (Telegram, Discord, PagerDuty) with rate limiting and priority handling.

use std::collections::HashMap;
use std::time::{Duration, Instant};

#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
pub enum AlertPriority {
    Info = 0,
    Warning = 1,
    Critical = 2,
    Emergency = 3,
}

#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
pub enum AlertChannel {
    Telegram,
    Discord,
    PagerDuty,
    Email,
}

#[derive(Debug, Clone)]
pub struct Alert {
    pub priority: AlertPriority,
    pub category: String,
    pub title: String,
    pub message: String,
    pub timestamp_ns: u64,
    pub dedup_key: Option<String>,
}

pub struct AlertManager {
    /// Rate limit per category (max alerts per minute)
    rate_limits: HashMap<String, (u32, Instant, u32)>, // (limit, window_start, count)
    /// Last alert time per dedup key (for suppression)
    dedup_cache: HashMap<String, Instant>,
    /// Dedup window duration
    dedup_window: Duration,
    /// Enabled channels
    enabled_channels: Vec<AlertChannel>,
    /// Alert queue for async dispatch
    queue: Vec<Alert>,
}

impl AlertManager {
    pub fn new() -> Self {
        Self {
            rate_limits: HashMap::new(),
            dedup_cache: HashMap::new(),
            dedup_window: Duration::from_secs(300), // 5 minute dedup
            enabled_channels: vec![AlertChannel::Telegram],
            queue: Vec::new(),
        }
    }
    
    /// Configure rate limit for a category
    pub fn set_rate_limit(&mut self, category: &str, max_per_minute: u32) {
        self.rate_limits.insert(
            category.to_string(),
            (max_per_minute, Instant::now(), 0)
        );
    }
    
    /// Queue an alert for dispatch
    pub fn alert(&mut self, alert: Alert) -> bool {
        // Check dedup
        if let Some(key) = &alert.dedup_key {
            if let Some(last) = self.dedup_cache.get(key) {
                if last.elapsed() < self.dedup_window {
                    return false; // Suppressed
                }
            }
            self.dedup_cache.insert(key.clone(), Instant::now());
        }
        
        // Check rate limit
        if let Some((limit, window_start, count)) = self.rate_limits.get_mut(&alert.category) {
            if window_start.elapsed() > Duration::from_secs(60) {
                *window_start = Instant::now();
                *count = 0;
            }
            if *count >= *limit {
                return false; // Rate limited
            }
            *count += 1;
        }
        
        self.queue.push(alert);
        true
    }
    
    /// Get pending alerts for dispatch
    pub fn drain_queue(&mut self) -> Vec<Alert> {
        std::mem::take(&mut self.queue)
    }
}

// Alert helper macros for use throughout codebase
#[macro_export]
macro_rules! alert_critical {
    ($mgr:expr, $cat:expr, $title:expr, $($arg:tt)*) => {
        $mgr.alert(Alert {
            priority: AlertPriority::Critical,
            category: $cat.to_string(),
            title: $title.to_string(),
            message: format!($($arg)*),
            timestamp_ns: std::time::SystemTime::now()
                .duration_since(std::time::UNIX_EPOCH)
                .unwrap_or_default()
                .as_nanos() as u64,
            dedup_key: Some(format!("{}:{}", $cat, $title)),
        })
    };
}
```

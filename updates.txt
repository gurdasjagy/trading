1. ARCHITECTURE OVERVIEW
CURRENT (USE_MULTI_EXCHANGE=off):
  Gate.io WS ──► SPSC Ring ──► Book Builder ──► Strategy ──► Execution (Gate.io only)

NEW (USE_MULTI_EXCHANGE=on):
  Gate.io WS  ──┐
  Binance WS  ──┼──► SPSC Ring (exchange_id tagged) ──► Book Builder (per-exchange books)
  Bybit WS    ──┘                                              │
                                                    ┌──────────┴──────────┐
                                                    │  Unified BBO Fusion │
                                                    └──────────┬──────────┘
                                                               │
                                                    ┌──────────┴──────────┐
                                                    │   Strategy Engine   │
                                                    └──────────┬──────────┘
                                                               │
                                                    ┌──────────┴──────────┐
                                                    │  MultiExchangeMgr   │
                                                    │  (best price route) │
                                                    └──────────┬──────────┘
                                                    ┌──────────┴──────────┐
                                                    │  Execution Router   │
                                                    │  Gate.io / Binance  │
                                                    │  / Bybit gateways   │
                                                    └─────────────────────┘

Key Design Decisions:

All 3 WS feeds push into the SAME SPSC ring buffer (tagged with exchange_id). This avoids tripling ring buffer memory and keeps the book builder loop simple.
The book builder maintains SEPARATE FlatOrderBook per exchange per symbol, PLUS a fused "unified BBO" that takes the best bid across all exchanges and best ask across all exchanges.
The strategy engine evaluates on the UNIFIED BBO (best prices across all exchanges).
The execution router queries MultiExchangeManager to determine which exchange has the best fill price for the signal direction, then routes to that exchange's gateway.
Position tracking is per-exchange. The MultiExchangeManager aggregates for portfolio-level risk.

//! Multi-Exchange Portfolio Manager.
//!
//! Tracks balances, positions, and BBO across Gate.io, Binance, and Bybit.
//! Provides unified portfolio view and smart order routing based on best
//! available price across all connected exchanges.
//!
//! Thread safety: All fields use atomic types or parking_lot locks.
//! Accessed from strategy thread (read), execution thread (write),
//! dashboard thread (read), and telemetry thread (read).

use std::collections::HashMap;
use std::sync::atomic::{AtomicBool, AtomicI64, AtomicU64, Ordering};
use std::sync::Arc;

use parking_lot::RwLock;
use tracing::{debug, info, warn};

use crate::config::exchange_id;
use crate::execution_gateway::Position;
use crate::fixed_point::FixedPrice;

// ---------------------------------------------------------------------------
// Per-Exchange BBO (Best Bid/Offer)
// ---------------------------------------------------------------------------

/// Atomic best-bid-offer for a single exchange + symbol pair.
/// Updated by the book builder thread, read by strategy and execution threads.
#[derive(Debug)]
pub struct AtomicBBO {
    pub best_bid_fp: AtomicI64,
    pub best_ask_fp: AtomicI64,
    pub bid_depth_usdt_fp: AtomicI64,
    pub ask_depth_usdt_fp: AtomicI64,
    pub spread_bps: AtomicI64,
    pub last_update_ns: AtomicU64,
}

impl AtomicBBO {
    pub fn new() -> Self {
        Self {
            best_bid_fp: AtomicI64::new(0),
            best_ask_fp: AtomicI64::new(0),
            bid_depth_usdt_fp: AtomicI64::new(0),
            ask_depth_usdt_fp: AtomicI64::new(0),
            spread_bps: AtomicI64::new(0),
            last_update_ns: AtomicU64::new(0),
        }
    }

    pub fn update(&self, bid: i64, ask: i64, bid_depth: i64, ask_depth: i64, spread: i64, ts: u64) {
        self.best_bid_fp.store(bid, Ordering::Relaxed);
        self.best_ask_fp.store(ask, Ordering::Relaxed);
        self.bid_depth_usdt_fp.store(bid_depth, Ordering::Relaxed);
        self.ask_depth_usdt_fp.store(ask_depth, Ordering::Relaxed);
        self.spread_bps.store(spread, Ordering::Relaxed);
        self.last_update_ns.store(ts, Ordering::Release);
    }

    /// Check if data is fresh (within 5 seconds).
    pub fn is_fresh(&self) -> bool {
        let now = crate::execution_gateway::now_ms() as u64 * 1_000_000;
        let last = self.last_update_ns.load(Ordering::Acquire);
        now.saturating_sub(last) < 5_000_000_000
    }

    pub fn best_bid(&self) -> i64 { self.best_bid_fp.load(Ordering::Relaxed) }
    pub fn best_ask(&self) -> i64 { self.best_ask_fp.load(Ordering::Relaxed) }
}

impl Default for AtomicBBO {
    fn default() -> Self { Self::new() }
}

/// Per-exchange balance tracking using atomics.
#[derive(Debug)]
pub struct ExchangeBalance {
    /// Available balance in USDT (x 1e8 fixed-point).
    pub available_fp: AtomicI64,
    /// Total equity in USDT (x 1e8 fixed-point).
    pub equity_fp: AtomicI64,
    /// Unrealized PnL in USDT (x 1e8 fixed-point).
    pub unrealized_pnl_fp: AtomicI64,
    /// Number of open positions on this exchange.
    pub position_count: AtomicU64,
    /// Whether this exchange is connected and operational.
    pub connected: AtomicBool,
    /// Last successful balance fetch timestamp (epoch seconds).
    pub last_sync_epoch: AtomicU64,
}

impl ExchangeBalance {
    pub fn new() -> Self {
        Self {
            available_fp: AtomicI64::new(0),
            equity_fp: AtomicI64::new(0),
            unrealized_pnl_fp: AtomicI64::new(0),
            position_count: AtomicU64::new(0),
            connected: AtomicBool::new(false),
            last_sync_epoch: AtomicU64::new(0),
        }
    }

    pub fn set_balance(&self, available: f64, equity: f64, unrealized: f64) {
        self.available_fp.store((available * 1e8) as i64, Ordering::Relaxed);
        self.equity_fp.store((equity * 1e8) as i64, Ordering::Relaxed);
        self.unrealized_pnl_fp.store((unrealized * 1e8) as i64, Ordering::Relaxed);
        let now = std::time::SystemTime::now()
            .duration_since(std::time::UNIX_EPOCH)
            .unwrap_or_default()
            .as_secs();
        self.last_sync_epoch.store(now, Ordering::Release);
    }

    pub fn available(&self) -> f64 { self.available_fp.load(Ordering::Relaxed) as f64 / 1e8 }
    pub fn equity(&self) -> f64 { self.equity_fp.load(Ordering::Relaxed) as f64 / 1e8 }
    pub fn unrealized_pnl(&self) -> f64 { self.unrealized_pnl_fp.load(Ordering::Relaxed) as f64 / 1e8 }
}

impl Default for ExchangeBalance {
    fn default() -> Self { Self::new() }
}

/// Per-exchange fee schedule (in basis points).
struct FeeSchedule {

    taker_bps: f64,

// ---------------------------------------------------------------------------
// MultiExchangeManager
// ---------------------------------------------------------------------------

/// Unified cross-exchange portfolio manager.
///
/// Thread-safe: uses atomics for scalar data and parking_lot::RwLock for
/// position vectors. Safe to access from strategy, execution, dashboard,
/// and telemetry threads simultaneously.
pub struct MultiExchangeManager {
    /// Per-exchange balances. Index: exchange_id (0=gateio, 1=binance, 2=bybit).
    pub balances: [ExchangeBalance; 3],

    /// Per-exchange, per-symbol BBO.
    /// Key: (exchange_id, symbol_id) -> AtomicBBO.
    /// Using RwLock<HashMap> because the set of symbols is fixed at startup.
    bbo_map: RwLock<HashMap<(u8, u16), Arc<AtomicBBO>>>,

    /// Per-exchange positions.
    /// Key: exchange_id -> Vec<Position>.
    positions: [RwLock<Vec<Position>>; 3],

    /// Fee schedules per exchange.
    fees: [FeeSchedule; 3],

    /// Whether multi-exchange mode is active.
    pub enabled: AtomicBool,
}

impl MultiExchangeManager {
    pub fn new() -> Self {
        Self {
            balances: [
                ExchangeBalance::new(),
                ExchangeBalance::new(),
                ExchangeBalance::new(),
            ],
            bbo_map: RwLock::new(HashMap::new()),
            positions: [
                RwLock::new(Vec::new()),
                RwLock::new(Vec::new()),
                RwLock::new(Vec::new()),
            ],
            fees: [
                FeeSchedule { maker_bps: -2.5, taker_bps: 7.5 },  // Gate.io
                FeeSchedule { maker_bps: 2.0, taker_bps: 4.0 },   // Binance
                FeeSchedule { maker_bps: 1.0, taker_bps: 6.0 },   // Bybit
            ],
            enabled: AtomicBool::new(false),
        }
    }

    /// Register a symbol for BBO tracking on all exchanges.
    pub fn register_symbol(&self, symbol_id: u16) {
        let mut map = self.bbo_map.write();
        for ex_id in [exchange_id::GATEIO, exchange_id::BINANCE, exchange_id::BYBIT] {
            map.entry((ex_id, symbol_id))
                .or_insert_with(|| Arc::new(AtomicBBO::new()));
        }
    }

    /// Update BBO for a specific exchange + symbol.
    /// Called by the book builder thread after processing orderbook updates.
    pub fn update_bbo(
        &self,
        ex_id: u8,
        symbol_id: u16,
        bid: i64,
        ask: i64,
        bid_depth: i64,
        ask_depth: i64,
        spread_bps: i64,
        timestamp_ns: u64,
    ) {
        let map = self.bbo_map.read();
        if let Some(bbo) = map.get(&(ex_id, symbol_id)) {
            bbo.update(bid, ask, bid_depth, ask_depth, spread_bps, timestamp_ns);
        }
    }

    /// Get the unified best bid across all exchanges for a symbol.
    /// Returns (price_fp, exchange_id) or None if no fresh data.
    pub fn unified_best_bid(&self, symbol_id: u16) -> Option<(i64, u8)> {
        let map = self.bbo_map.read();
        let mut best: Option<(i64, u8)> = None;
        for ex_id in [exchange_id::GATEIO, exchange_id::BINANCE, exchange_id::BYBIT] {
            if let Some(bbo) = map.get(&(ex_id, symbol_id)) {
                if bbo.is_fresh() {
                    let bid = bbo.best_bid();
                    if bid > 0 {
                        match best {
                            None => best = Some((bid, ex_id)),
                            Some((b, _)) if bid > b => best = Some((bid, ex_id)),
                            _ => {}
                        }
                    }
                }
            }
        }
        best
    }

    /// Get the unified best ask across all exchanges for a symbol.
    /// Returns (price_fp, exchange_id) or None if no fresh data.
    pub fn unified_best_ask(&self, symbol_id: u16) -> Option<(i64, u8)> {
        let map = self.bbo_map.read();
        let mut best: Option<(i64, u8)> = None;
        for ex_id in [exchange_id::GATEIO, exchange_id::BINANCE, exchange_id::BYBIT] {
            if let Some(bbo) = map.get(&(ex_id, symbol_id)) {
                if bbo.is_fresh() {
                    let ask = bbo.best_ask();
                    if ask > 0 {
                        match best {
                            None => best = Some((ask, ex_id)),
                            Some((a, _)) if ask < a => best = Some((ask, ex_id)),
                            _ => {}
                        }
                    }
                }
            }
        }
        best
    }

    /// Determine the best exchange to route an order to.
    ///
    /// For BUY orders: route to the exchange with the lowest ask (cheapest to buy).
    /// For SELL orders: route to the exchange with the highest bid (best sell price).
    ///
    /// Factors in fees: total_cost = price_disadvantage + taker_fee.
    /// Falls back to Gate.io (exchange_id=0) if no multi-exchange data available.
    pub fn best_exchange_for_order(
        &self,
        symbol_id: u16,
        is_buy: bool,
        is_maker: bool,
    ) -> MultiExchangeRoutingDecision {
        if !self.enabled.load(Ordering::Relaxed) {
            return MultiExchangeRoutingDecision {
                exchange_id: exchange_id::GATEIO,
                price: 0.0,
                estimated_cost_bps: 0.0,
                reason: "single-exchange mode",
            };
        }

        let map = self.bbo_map.read();
        let mut best_cost = f64::MAX;
        let mut best_ex = exchange_id::GATEIO;
        let mut best_price = 0.0;
        let mut best_reason = "fallback to primary";

        for ex_id in [exchange_id::GATEIO, exchange_id::BINANCE, exchange_id::BYBIT] {
            if let Some(bbo) = map.get(&(ex_id, symbol_id)) {
                if !bbo.is_fresh() {
                    continue;
                }
                // Check exchange is connected
                if !self.balances[ex_id as usize].connected.load(Ordering::Relaxed) {
                    continue;
                }

                let price_fp = if is_buy {
                    bbo.best_ask()
                } else {
                    bbo.best_bid()
                };

                if price_fp <= 0 {
                    continue;
                }

                let price_f64 = FixedPrice(price_fp).to_f64();
                let fee_bps = if is_maker {
                    self.fees[ex_id as usize].maker_bps
                } else {
                    self.fees[ex_id as usize].taker_bps
                };

                // Cost model: fee + half-spread
                let spread = bbo.spread_bps.load(Ordering::Relaxed) as f64;
                let total_cost = fee_bps + spread / 2.0;

                if total_cost < best_cost {
                    best_cost = total_cost;
                    best_ex = ex_id;
                    best_price = price_f64;
                    best_reason = match ex_id {
                        exchange_id::GATEIO => "best cost on Gate.io",
                        exchange_id::BINANCE => "best cost on Binance",
                        exchange_id::BYBIT => "best cost on Bybit",
                        _ => "unknown",
                    };
                }
            }
        }

        MultiExchangeRoutingDecision {
            exchange_id: best_ex,
            price: best_price,
            estimated_cost_bps: best_cost,
            reason: best_reason,
        }
    }

    /// Update positions for an exchange (called after REST position sync).

/// Main WS ingestion loop for Binance Futures.
/// Runs on a dedicated OS thread with its own tokio runtime.
pub fn ws_ingestion_loop_binance(
    ring: &'static SpscRingBuffer<RawBookUpdate, 65536>,

    symbols: Vec<String>,
    registry: Arc<SymbolRegistry>,

        .enable_all()
        .build()
        .expect("Failed to build tokio runtime for ws-binance");

    rt.block_on(async {

        let mut drop_count: u64 = 0;
        let mut msg_count: u64 = 0;

        loop {
            let base_url = if testnet { BINANCE_WS_TESTNET } else { BINANCE_WS_LIVE };

            // Build combined stream URL: /stream?streams=btcusdt@depth20@100ms/btcusdt@aggTrade
            let streams: Vec<String> = symbols.iter().map(|s| {
                let sym = to_binance_stream_symbol(s);
                format!("{}@depth20@100ms/{}@aggTrade", sym, sym)
            }).collect();

            ).await {
                Ok(()) => {
                    info!("[ws-binance] Connection closed normally, reconnecting in 1s");
                    backoff_secs = 1;
                }
                Err(e) => {
                    let jitter = (backoff_secs as f64 * 0.2) as u64;
                    let sleep_secs = backoff_secs.saturating_add(jitter);
                    warn!(
                        "[ws-binance] Error: {}. Reconnecting in {}s (drops={}, msgs={})",
                        e, sleep_secs, drop_count, msg_count
                    );
                    backoff_secs = (backoff_secs * 2).min(60);
                }
            }

            tokio::time::sleep(Duration::from_secs(backoff_secs)).await;
        }
    });
}

async fn connect_and_ingest_binance(
    url: &str,
    ring: &'static SpscRingBuffer<RawBookUpdate, 65536>,
    trades_ring: &'static SpscRingBuffer<TradeEvent, 16384>,
    registry: &SymbolRegistry,
    drop_count: &mut u64,
    msg_count: &mut u64,
) -> Result<(), Box<dyn std::error::Error + Send + Sync>> {
    let (ws_stream, _) = connect_async(url).await?;
    let (mut _write, mut read) = ws_stream.split();

    info!("[ws-binance] Connected to Binance Futures stream");

    let now_ns = || -> u64 {
        std::time::SystemTime::now()
            .duration_since(std::time::UNIX_EPOCH)
            .unwrap_or_default()
            .as_nanos() as u64
    };

    while let Some(msg) = read.next().await {
        match msg? {
            Message::Text(text) => {
                let recv_ns = now_ns();
                *msg_count += 1;

                // Binance combined stream format: {"stream":"btcusdt@depth20@100ms","data":{...}}
                let parsed: serde_json::Value = match serde_json::from_str(&text) {
                    Ok(v) => v,
                    Err(_) => continue,
                };

                let stream = parsed.get("stream").and_then(|v| v.as_str()).unwrap_or("");
                let data = match parsed.get("data") {
                    Some(d) => d,
                    None => continue,
                };

                if stream.contains("@depth") {
                    // Orderbook depth update
                    // Extract symbol from stream name: "btcusdt@depth20@100ms" -> "BTCUSDT"
                    let binance_sym = stream.split('@').next().unwrap_or("").to_uppercase();
                    // Convert to our format: BTCUSDT -> BTC_USDT
                    let our_sym = binance_to_internal_symbol(&binance_sym);
                    let sym_id = registry.get_id(&our_sym);
                    if sym_id == 0 { continue; }

                    // Parse bids: [[price, qty], ...]
                    if let Some(bids) = data.get("b").or(data.get("bids")).and_then(|v| v.as_array()) {
                        for level in bids {
                            if let Some(arr) = level.as_array() {
                                if arr.len() >= 2 {
                                    let price = arr[0].as_str().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
                                    let qty = arr[1].as_str().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
                                    if price > 0.0 {
                                        let update = RawBookUpdate {
                                            symbol_id: sym_id,
                                            side: spsc::side::BID,
                                            update_type: spsc::update_type::DELTA,
                                            exchange_id: spsc::exchange_id::BINANCE,
                                            _pad: [0; 3],
                                            price: FixedPrice::from_f64(price).raw(),
                                            qty: FixedQty::from_f64(qty).raw(),
                                            sequence: *msg_count,
                                            recv_ns,
                                            snapshot_count: 0,
                                            _pad2: [0; 4],
                                        };
                                        if !ring.try_push(update) {
                                            *drop_count += 1;
                                        }
                                    }
                                }
                            }
                        }
                    }

                    // Parse asks
                    if let Some(asks) = data.get("a").or(data.get("asks")).and_then(|v| v.as_array()) {
                        for level in asks {
                            if let Some(arr) = level.as_array() {
                                if arr.len() >= 2 {
                                    let price = arr[0].as_str().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
                                    let qty = arr[1].as_str().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
                                    if price > 0.0 {
                                        let update = RawBookUpdate {
                                            symbol_id: sym_id,
                                            side: spsc::side::ASK,
                                            update_type: spsc::update_type::DELTA,
                                            exchange_id: spsc::exchange_id::BINANCE,
                                            _pad: [0; 3],
                                            price: FixedPrice::from_f64(price).raw(),
                                            qty: FixedQty::from_f64(qty).raw(),
                                            sequence: *msg_count,
                                            recv_ns,
                                            snapshot_count: 0,
                                            _pad2: [0; 4],
                                        };
                                        if !ring.try_push(update) {
                                            *drop_count += 1;
                                        }
                                    }
                                }
                            }
                        }
                    }
                } else if stream.contains("@aggTrade") {
                    // Aggregate trade event
                    let binance_sym = stream.split('@').next().unwrap_or("").to_uppercase();
                    let our_sym = binance_to_internal_symbol(&binance_sym);
                    let sym_id = registry.get_id(&our_sym);
                    if sym_id == 0 { continue; }

                    let price = data.get("p").and_then(|v| v.as_str()).and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
                    let qty = data.get("q").and_then(|v| v.as_str()).and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
                    let is_buyer_maker = data.get("m").and_then(|v| v.as_bool()).unwrap_or(false);

                    if price > 0.0 && qty > 0.0 {
                        let side = if is_buyer_maker { 1u8 } else { 0u8 }; // m=true means seller is taker (sell)
                        let trade_event = TradeEvent {
                            symbol_id: sym_id,
                            side,
                            _pad: [0; 5],
                            price: FixedPrice::from_f64(price).raw(),
                            qty: FixedQty::from_f64(qty).raw(),
                            recv_ns,
                            sequence: *msg_count,
                        };
                        if !trades_ring.try_push(trade_event) {
                            *drop_count += 1;
                        }
                    }
                }
            }
            Message::Ping(data) => {
                // Binance sends pings; we need to respond with pong
                // But since we split the stream, we can't easily send pong here.
                // tokio-tungstenite auto-responds to pings by default.
                debug!("[ws-binance] Received ping ({} bytes)", data.len());
            }
            Message::Close(_) => {
                info!("[ws-binance] Received Close frame");
                return Ok(());
            }
            _ => {}
        }
    }

    Ok(())
}

                    Some(Ok(Message::Close(_))) => {
                        info!("[ws-bybit] Received Close frame");
                        return Ok(());
                    }
                    Some(Err(e)) => return Err(Box::new(e)),
                    None => return Ok(()),
                    _ => {}
                }
            }
            _ = ping_interval.tick() => {
                let ping_msg = serde_json::json!({"op": "ping"});
                if let Err(e) = write.send(Message::Text(ping_msg.to_string())).await {
                    warn!("[ws-bybit] Ping send failed: {}", e);
                    return Err(Box::new(e));
                }
            }
        }
    }
}
